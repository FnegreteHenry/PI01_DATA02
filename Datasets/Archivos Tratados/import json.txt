import json
import pandas as pd

with open('constructors2.json',encoding='UTF-8') as reader:
    constructors = json.load(reader)
df = pd.DataFrame(constructors)
df.set_index(df.constructorId,inplace=True)
df.drop(columns=['constructorId'],inplace=True)
df.head(8)
